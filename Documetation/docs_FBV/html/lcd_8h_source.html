00244 void lcd_cursor (char row, char column)
00245 {
00246         switch (row) {
00247                 case 1: lcd_wr_command (0x80 + column - 1); break;
00248                 case 2: lcd_wr_command (0xc0 + column - 1); break;
00249                 case 3: lcd_wr_command (0x94 + column - 1); break;
00250                 case 4: lcd_wr_command (0xd4 + column - 1); break;
00251                 default: break;
00252         }
00253 }
